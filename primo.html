يدعم نظام Primo ERP

النسخ الاحتياطي الأتوماتيكي يومياً أو إسبوعياً او حسب الإختيار ويعتمد على تحـــــديد صلاحيــــــات المستخدمين مـــما يتيح تقسييم الأعمال بشكل منظم وإطلاع كل موظف عـــلى الجزء الخاص به فقط كمـا يدعم اللغــــة العربية واللغة الإنجـــليزية ويعمل على خوادم سحابية أو خوادم داخلية

يوفر النظام شهـــــــادة تأمين بيــــــــــــانات للنسخة السحابية كما يوفـــــر إمكانية إستيراد وتصدير البياـــــــنات بصيـغ مختلفة مثل PDF أو Excel

مكونات نظام Primo ERP

يحتوى النظام على إدارات مختلفة تمكـنك من تنسيــــــق وإدارة أعمـــــالك في مختـــلف المجالات فهو يحتـــوي على إدارة حسابات عامة ونقدية, أدارة للعملاء والمــــوردين, إدارة المـــخازن, إدارة المشتريات, إدارة المبيعــات, إدارة الموارد البشـــرية, إدارة الأصـــــول الثابتة والصيــــانة, إدارة نقاط البيع, إدارة العيادات والمؤسسات الطــــــبية, إدارة النــــــادي الرياضي و إدارة الصيدليات وإدارة الرسائل القصيرة...وسوف نتطرق لكل إدارة من إدارات نظام

النظام المحاسبي وإدارة النقدية

يـعتمد نظــام المحاسبــــي علـي قوائــــم مالــــــية ديناميكـية متـطورة مرتبــطة بكــــــــل أدارات النظام المختلفة ويوفر النظام دليل حســــابات شجـــري لا نهائــي المسـتـــويـــات مــــــع توافــــــر وسائـــــــــل مساعـدة حيث يوفـــــر تصنيف لدليــــل الحسابـــــات استرشـــــــادي ويعـــالج اخطـــاء المستخــــــدم عنــد انشـــاء دلــــيل الحسابــــات مــمـــا يتيــــح استــخراج الحســـــــابات الخــــتامية ديناميكـــــيا و يوفر نظـــام النقديــــة كـــــشف حـــــــساب للخــــزائن والبنــــوك بالعمـلات المختــــلفة مــــع إمكانــية تحـــديد الحـد الأقصي والحد الأدني لرصـــــــيد الخــزائن والبنـــــوك و إمكانــية السحب علــــي المكشوف بصلاحيــــات

نظام المخازن

نظـــــام مخازن متـــــكامــل يعمل بشكل منفصــــل أو متكامــــل مــــــــــع المشتــــريـات والمبـــيعـــات والحسابــات و يتميــــــز النظام بتعـــــريف أصنـــــاف شجري لا نهائـــــي المستويـــــات مــع تعـدد وحدات صرف الأصناف لكل صنف كما يدعم النظـام تعــــدد المخازن والصلاحيات علــــي كل مخـــزن وإمكانيـــــة تخصيص صلاحيات محـــددة لكل مستخدم علي نفس المــــخزن خلال فتـــــرة معينة و يوفر النـظام متابعة دقيقة للعيــــون المخزنية وأرصــــدة العيـون المخزنية وتحديد مواقع الأصناف دخل كل مخزن

إدارة العملاء والموردين

تعـــــريف العـملاء والمـــــــوردين بإسلوب شجري لا نهائي مع تحــــديد شـــــجري ممــــــاثــل للمندوبيـــن فهــــو نظام مـــرن وميسر للتـــــعامل مــــع الضرائب وإنشاء نســــــب وشرائح ضريبية مختلفـة للعمــــــلاء والموردين كما يدعم توفيـر نسب وشـرائح خصومات مختــــلفة ومتنوعـة للعملاء والأصنـــاف وللموردين والأصناف . نظام العملاء والموردون مرتبــــط بإدارة المشــتريات والمبيعات والنقديــة ويـــــدعم النــــظام تعدد العملات وكشـــــــف حـــساب بعملة العميـــل أو المورد و يــدعم إنشاء حــــد إئتمــاني محدد المدة و يوفر إسلوب لقياس أعـــمار الديون كـــــما يدعــــم إنشاء وتعديل خطط السداد والتحصيل

إدارة المشتريات

نظام مشتـــريات داينمك مرتبط بالعملاء والموردين وإدارة المخـــازن والأصنـــــاف الخــــــدمية كما أنــــه مرتبط بالحسابات العامة وعند إنشاء أوامر الـشراء يتم تحديد فتـــرة صلاحيـة لأمر الشـــــراء ولا يحـــــــق لأي مستخدم إصدار أمر شراء بدون الحصول عـلي صلاحية لذلك . يدعم نظام المشتريات الإضافات والخصومات عـــلي الأصنــــاف وعلــــي إجمــــالـي أمـــــر الشـراء أو فاتورة المشــــــتريات كـــــما يدعــم النظـــام دورة المشتريات كامـلة بـداية من أمر الشراء حتي استلام الأصنـاف بالمخـزن وكل ما يتبـــع ذلك من حركات

إدارة المبيعات

نظام المبيعات داينمـك مــــرتبط بالعملاء الموردين وإدارة المــــخازن والأصنــــاف الخدمـــيـة كما أنــــه مرتبط بالحسابات العامة و عند إنشاء أوامر البيــع يتم تحديد فتــــــرة صلاحيــة لأمر البـــــــيع ولا يحـــق لأي مستــــخـدم اصـــدار امــر بيع بــدون الحصـــول عـــلي صلاحية لذلك كــــــما يدعم نظام المبيعات لإضافات والخصومــات علــــي الأصــــــنـاف وعلــــي إجمالي أمــــــر البيـــــع أو فــــاتورة المــــــــبيعات و معــالجة الدفعات المــقدمة علـــي أوامــــر البيــــع وخطـــــط تحصيــــل أوامر البـــــيع و إمكانية التعديــــــل علــــيها بصلاحيات و يدعــــم النظــــــام دورة مبيعات كاملة بداية من أمـــر البيع وحتي تسليم الأصــــناف مــــن المخـــــزن وكل ما يتبع ذلك من حركات

إدارة الموارد البشرية

يدعم نظام إدارة الموارد البشرية العـــــمل منفـصلاً كنظام قائم بذاته أو كجـــزء مترابـط مـــــع كامــــــل نظام ويقوم بإنشاء قيـــود المرتبات والاستحقاقات الشهرية والسنويـة بــــشكل أتوماتيكي فـــي قيود اليومية العامــة كمـا يتكون نظـــــام الموارد البشرية من ثلاثة مكــــونات رئيســــــــية (الهيــكل التنظيمي للمؤسســات – إدارة الــــوقت – إدارة المـــــــرتبــــات) ويعتمد الهيكل التنــــظيمي للمؤسسات والشركات على هيـــكل شجري للإدارة الرئيســـــــية والأقــسام الفرعـــــية والوظـــــــائف لكـــــل إدارة بالإضافة إلى التوصيف الوظيفي

إدارة الأصول الثابتة

يدعــم نظــــام الأصول الثابتة العمـل منفصلا كنظام قائم بذاته أو كـــجزء مترابط مع كامل نظام ويقوم بإنشاء القيــــــود المحاسبية بشكل مباشـر في نظام الحسابات العامة و يوفر النظام تعـــريف شجــــري لا نهائي للأصول الثابتة والصيانة مـع تقســيم الأصــول الثابتـة حــــسب نوعها (أراضي - ممــتلـكات - آلات - عـدد ...الخ) كما يدعم النظام المتابعة المالية للأصول الثابتة مـــن تاريـــخ شراء الأصـــل وإحتســاب تكلفــــة الحصـول عليـــه مرورا بإحتســـاب إستـــهلاك الأـصول خلال العــمر الإنتاجــــــــي للأصل حتـي الوصـول إلي مرحلة بيعـــه أو تكهــــينه بإستخــدام أسـاليب وطرق الإستهـلاك المحاســبية (القســط الثابت – القــــسط المتناقص)

نظام الإدارة الطبية للعيادات المتخصصة والمجمعات الطبية

خاص بإدارة الأطباء وورديات العمــل وربط الأطـــــباء بالعيادات وإدارة الغرف والخدمات من حيث التصنيف العـام والأسعار والعــروض مع تحليل إيـرادات الأطباء والعيادات والفروع والغرف كما يوفر نظام تكـــــويد الفــــــــــواتير ومصـــــــدر كل فاتـــــورة (المتحصلات-المدفوعــات-التــــسويات) و كشـــــــــف حساب لكل مريض وتحليل الخصومات التي يحصل عليها المرضــى كـما يقوم بربط النظـــام الطــــبي بالحسابات العامة وإدارة النقــدية وإنشاء القيود المحاســبية أتوماتيكياً بالحسابات العامـــــة ويــــــوفر ربط النظام الطبــــــي بالمخازن وعــــمل مخزن فرعي لكل عيادة

إدارة نقاط البيع

يــدعم النظام تعــــدد الصلاحيات باسم المســــــتخدم وكـــلمة مرور كمــــــــا يدعــــم شاشــــــات اللمــــس و تقسيم الاصناف الــي مجــــــموعات مـــع امكـانية عرض صــورة الصـــنف و امكــــــــانية البيع لاكثر مــن عميل في نفس الـــوقت كما يدعم اضافـــــة بيانات تفصيلية عــــن العــــــملاء و يدعــــــــم طــرق الدفـــع المخــــتلفة نقدي وتحويل شبــــــــكي كـــــما يمكن تعليــــق العـــــمل اثنـــاء الورديـــــة و تسجيل مرتجـــع المبيعات و تسجيل المصروفات النثرية اليومـــــية من خلال نقطة البيع و عند اقفال وردية الـبائع يتم اصدار تقرير بيومية المبيعات وتحويل المــــبالغ النقدية الي الخزينة الرئيسية

إدارة النادي الرياضي

لوحة تحكم رئيسية لإدارة إعدادات النــــادي الرياضي مثل إدارة مديري وموظفي النادي الرياضي (مــــدير النادي - مديرر الاستـــــــقبال -مدير التدريب - اخصائي التغذية - الـسوبر ادمن - مدير التشغيــــــل والصيانة - مدير تسويق - اداري) تخصيـــــــــص موظـفي النادي الرياضي حسب تتخــصصهم لتحديد الصلاحيات وادارة المستخدمين مـــــــتابعة وادارة التحـكم في استقبال النادي الرياضــــي من خـــــــلال نقاط بيع الخــــدمات والإشتركات والمنــتجات الرياضية تنـــوع طريق الدفع في نقاط البيع (نقدي - شبكة - مدي - ماستر كارد - اس تي سي - محفــــظة الكــــترونية - نقـاط العميل FIT – الكوبونات) إدارة الخدمــــــات والإشــــــتراكات تصنيفات رئيسية وتصنيفات فــــرعية (تغذية مــــــساج تدريب شخصي اشتراكات استشارة رياضية خدمات اضافية بالإضافة الي الأدوات والمستلزومات الرياضية) الادارة الذكية للخدمات والاشتركات (تشغيل - ايقاف - تعديل) مع امكانية تفعيل الخدمات والاشتراكات خلال تاريخ محدد واظهارها في نقاط البيع خلال فترة محددة اعتماد الخدمات والاشتراكات من خلال مدير النادي الرياضي مع امكانية ارفاق خطاب الاعتماد ارفاق عقد الخدمة وسهولة استدعائة للعميل للتوقيع علية عند بيع الخدمة او تفعيل الخدمة امكانية تسعير المنتجات والاشتراكات والمنتجات باسعار مختلفة بين المشتركين وغير المشتركين وتغير سعر الخدمة من فرع لفرع اخر طرق مختلفة لمبيعات الخدمات كعدد ايام او عدد جلسات وتحديد مدة الخدمات او عدد الجلسات وعدد ايام الدخول أمكانية ربط الخدمات والاشتراكات ببوابات دخول محددة سواء داخل الفرع الواحد مثل الصالات بوابة الدخول الرئيسية بوبات الدخول علي الصالات المختلفة او اكثر من فرع وخلال توقيتات دخول محددة لكل بوابه تفعيل الخدمات والاشتراكات بعد اشتراك العميل وبالتالي تفعيل بصمة دخول العميل علي البوابة المرتبطة بالخدمة من خلال موظف الاستقبال او تفعيل الخدمة آليا بعد فترة محددة يتم تحديدها عند انشاء الاشتراك او الخدمة تحديد الخدمات التي تتطلب وجود مدرب أو لا تحتاج مدرب مع إمكانية تحديد الخدمات واشتراكات التدريب الشخصي PT من وجهة النظر التسويقة يمكن تحديد الخدمات والاشتركات التي تستحق عمولة علي التدريب أو تستحق عمولة علي التسويق أولا تستحق اي عمولة وكثير من الخدمات الأخري
